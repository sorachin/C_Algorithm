#include<stdio.h>
#include<stdlib.h>

//int *parr;
void insert_sort(int *parr, int n) //n = 배열의 원소 개수
{
	int minLoc, temp;
	for (int step = 1; step < n; step++)
	{
		minLoc = step;
		for (int i = step; i >=  0; i--)
		{
			if (*(parr + step)<*(parr+i))
			{
				minLoc = i;
			}
			// 시간 단축 자리를 찾으면(앞부분이 정렬되있으므로) 
			else if(*(parr + step) > *(parr + i))
			{
				break;
			}
		}
		if (minLoc != step)
		{
			temp = *(parr + step);
			// 오류 부분
			// minLoc -> step 순으로 커지기 시작하면 모두 같은 값으로 바뀌므로 
			// 뒤에서부터 옮겨야함
			for (int j = step-1; j >= minLoc; j--)
			{
				*(parr + j + 1) = *(parr + j);
			}
			*(parr + minLoc) = temp;
		}
	}
}
int main()
{
	int n;
	scanf_s("%d", &n);
	int *parr;
	parr = (int*)malloc(n);

	//숫자 n번 입력받기
	for (int step = 0; step < n; step++)
	{
		scanf_s("%d", parr + step);
	}
	insert_sort(parr, n);
	for (int step = 0; step < n; step++)
	{
		printf(" %d", *(parr + step));
	}
	printf("\n");
}
