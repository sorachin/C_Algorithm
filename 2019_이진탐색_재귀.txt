#include<stdio.h>
#include<stdlib.h>
int *parr, K;

int findElement(int left, int right)
{
	int mid = (left + right) / 2;
	if (K < parr[left]) return left - 1;
	else if (left == right)
	{
		if (K >= parr[right]) return right;
		else return right - 1;
	}
	else if (K < parr[mid])  findElement(left, mid - 1);
	else if (K > parr[mid])  findElement(mid + 1, right);
}

int main()
{
	int N;
	int result;
	scanf_s("%d %d", &N, &K);

	parr = (int *)malloc(sizeof(int)*N);
	for (int step = 0; step < N; step++) scanf_s("%d", &parr[step]);
	
	result = findElement(0, N - 1);
	printf(" %d", result);
}
